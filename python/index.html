#import random library
import random

#function to display the player's stats
def print_player(player):
    print("Player stats:")
    for key in sorted(player):
        print(key + " is " + str(player[key]))

#function to calculate experience earned
def compute_experience(player, damage):
    player["experience"] += random.randrange(0, damage*2)

#function to deal damage to the player
def take_damage(player, damage):
    player["health"] -= damage
    compute_experience(player, damage)
    #check if player is dead
    if player["health"] <= 0:
        player["alive"] = False

#main game begins
player = {"name": "Sheldor", "title": "The Conqueror", "health": 100, "experience": 0, "mana": 10, "alive": True, "inventory": []}
player["inventory"].append("Sword of Azeroth")
compute_experience(player, 100)
while player["alive"]:
    print_player(player)
    take_damage(player, 5)
print("player dead")
